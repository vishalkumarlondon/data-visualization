#run a final OLS model
model_freq <- lm(log(`airbnb_freq`) ~ 
                    `bame_p` +
                    `nonUK` +
                    `education` +
                    `income` +
                    `house_mortg` +
                    `house_price` +
                    `culture_freq` +
                    `culture_rating`+
                    `InnerOuter`,
                  data = londonLSOAProfiles, na.action=na.exclude)

summary(model_freq)
